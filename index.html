// 你的專屬 Google Drive 相簿資料夾 ID var RECEIPT_FOLDER_ID = "1eSXqLmAWizWa_QpZ-jqklSSqaAFiDGYi"; function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); // 給 iPhone 捷徑讀取的隱藏推播通道 if (e.parameter && e.parameter.action === "checkTodos") { var todos = getSheetData(ss.getSheetByName("todos")); var urgentTasks = []; for (var i = 0; i < todos.length; i++) { var t = todos[i]; if (t.completed !== true && t.completed !== "true") { if (t.text.indexOf("繳費") !== -1 || t.text.indexOf("急") !== -1 || t.text.indexOf("卡") !== -1) { urgentTasks.push(t.text); } } } if (urgentTasks.length > 0) { var msg = "⚠️ 帳本提醒：\n今天有重要事項尚未完成：\n- " + urgentTasks.join("\n- "); return ContentService.createTextOutput(msg).setMimeType(ContentService.MimeType.TEXT); } else { return ContentService.createTextOutput("無重要待辦").setMimeType(ContentService.MimeType.TEXT); } } var result = { transactions: getSheetData(ss.getSheetByName("transactions")), todos: getSheetData(ss.getSheetByName("todos")), recurring_rules: getSheetData(ss.getSheetByName("recurring_rules")) }; return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var params = JSON.parse(e.postData.contents); var action = params.action; if (action === "addTransaction") { var sheet = ss.getSheetByName("transactions"); // 支援批次寫入陣列（用於跨帳本轉帳） var txs = Array.isArray(params.data) ? params.data : [params.data]; txs.forEach(function(t) { var finalImageUrl = ""; // 圖片上傳目前僅處理單筆夾帶 if (!Array.isArray(params.data) && params.imageData && params.imageData.base64) { var folder = DriveApp.getFolderById(RECEIPT_FOLDER_ID); var imageBlob = Utilities.newBlob(Utilities.base64Decode(params.imageData.base64), params.imageData.mimeType, "Receipt_" + t.id + "." + params.imageData.mimeType.split('/')[1]); var file = folder.createFile(imageBlob); finalImageUrl = "https://drive.google.com/uc?export=view&id=" + file.getId(); } // 新增寫入第 13 欄位：ledger (帳本歸屬) sheet.appendRow([t.id, t.type, t.date, t.desc, t.amount, t.payer, t.splitRatio, t.category, t.note, finalImageUrl, t.isSettled || false, t.uploader || 'System', t.ledger || 'shared']); }); } else if (action === "archive") { var sheet = ss.getSheetByName("transactions"); var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { // 只結清 shared 帳本裡的代墊款 var ledger = data[i][12] || 'shared'; if (data[i][1] === "expense" && data[i][5] !== "fund" && ledger === 'shared') { sheet.getRange(i + 1, 11).setValue(true); } } } else if (action === "deleteTransaction") { deleteRowById(ss.getSheetByName("transactions"), params.data.id); } else if (action === "addRecurringRule") { var sheet = ss.getSheetByName("recurring_rules"); var r = params.data; sheet.appendRow([r.id, r.desc, r.amount, r.payer, r.splitRatio, r.category, r.rule, r.ledger || 'shared']); } else if (action === "deleteRecurringRule") { deleteRowById(ss.getSheetByName("recurring_rules"), params.data.id); } else if (action === "addTodo") { var sheet = ss.getSheetByName("todos"); sheet.appendRow([params.data.id, params.data.text, false]); } else if (action === "toggleTodo") { var sheet = ss.getSheetByName("todos"); var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (data[i][0].toString() === params.data.id.toString()) { sheet.getRange(i + 1, 3).setValue(params.data.completed); break; } } } else if (action === "deleteTodo") { deleteRowById(ss.getSheetByName("todos"), params.data.id); } return ContentService.createTextOutput(JSON.stringify({status: "success"})).setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({status: "error", message: error.toString()})).setMimeType(ContentService.MimeType.JSON); } } function processRecurringExpenses() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var ruleSheet = ss.getSheetByName("recurring_rules"); if (!ruleSheet) return; var txSheet = ss.getSheetByName("transactions"); var rules = getSheetData(ruleSheet); var today = new Date(); var currentDay = today.getDate(); var currentWeekday = today.getDay(); var currentMonth = today.getMonth() + 1; var formattedDate = Utilities.formatDate(today, Session.getScriptTimeZone(), "yyyy-MM-dd"); for (var i = 0; i < rules.length; i++) { var r = rules[i]; var ruleParts = r.rule.split('_'); var type = ruleParts[0]; var values = ruleParts.length > 1 ? ruleParts[1].split(',').map(Number) : []; var shouldTrigger = false; if (type === "daily") shouldTrigger = true; else if (type === "weekly" && values.includes(currentWeekday)) shouldTrigger = true; else if (type === "monthly" && values.includes(currentDay)) shouldTrigger = true; else if (type === "yearly" && values.includes(currentMonth) && currentDay === 1) shouldTrigger = true; if (shouldTrigger) { txSheet.appendRow([ new Date().getTime() + i, "expense", formattedDate, "[自動] " + r.desc, r.amount, r.payer, r.splitRatio, r.category, "", "", false, "System", r.ledger || 'shared' ]); } } } function getSheetData(sheet) { if (!sheet) return []; var rows = sheet.getDataRange().getValues(); if (rows.length < 2) return []; var headers = rows[0]; var data = []; for (var i = 1; i < rows.length; i++) { var obj = {}; for (var j = 0; j < headers.length; j++) { if(headers[j] === 'date' && rows[i][j] instanceof Date) { obj[headers[j]] = Utilities.formatDate(rows[i][j], Session.getScriptTimeZone(), "yyyy-MM-dd"); } else { obj[headers[j]] = rows[i][j]; } } data.push(obj); } return data; } function deleteRowById(sheet, id) { if(!sheet) return; var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (data[i][0].toString() === id.toString()) { sheet.deleteRow(i + 1); break; } } }
